Sitemap example - Bilingual Site

<?xml version="1.0" encoding="UTF-8" ?>
<urlset xmlns="http://sitemaps.org/schemas/sitemap/0.9"
    xmlns:xhtml="html://www.w3.org/1999/xhtml">
    http://domain.com/about/
    <xhtml:link
      rel="alternate"
      hreflang="en-ca"
      href="http://domain.com/en-ca/about/"
      />
    <xhtml:link
      rel="alternate"
      hreflang="fr-ca"
      href="http://domain.com/fr-ca/about/"
      />
    <xhtml:link
      rel="alternate"
      hreflang="en"
      href="http://domain.com/en/about/"
      />

//and then another URL, and on and on
